Json常用的处理方法
g.extend Function(Boolen,Object/Array,Object/Array,...) 
  2.                  //*------JSON扩展，后面对象继承前面对象 --------*/ 
  3.                  //例如1：   
  4.                 var str1 = { "id": "1", "name": "gly" };
  5.                 var str0 = { "name": "zhoujingli", "sex": "男" };
  6.                 var str2 = g.extend(str1, str0);
  7.                 console.log(str2); //str2就是'{"id":"1","name":"zhoujingli","sex":"男"}';
  9.                  //例如2： 
  10.                 var str1 = [{ "id": "1", "name": "gly" }, { "id": "1", "name": "gly" }];
  11.                 var str0 = [{ "name": "zhoujingli", "sex": "男" }];
  12.                 var str2 = g.extend(str1, str0);
  13.                 console.log(str2); //str2就是[{"id":"1","name":"zhoujingli","sex":"男"},{"id":"1","name":"gly"}];
  15.                  //例如3： 
  16.                 var str1 = [{ "id": "1", "name": { "id": "32", "name": "ffff" } }];
  17.                 var str0 = [{ "name": "zhoujingli", "sex": "男" }];
  18.                 var str2 = g.extend(str1, str0);
  19.                 console.log(str2); //str2就是[{"id":"1","name":"zhoujingli","sex":"男"}];
  21.                  //例如4： 
  22.                 var str1 = { "id": "1", "name": "gly" };
  23.                 var str0 = { "obj": { "id": "32", "name": "ffff" } };
  24.                 var str2 = g.extend(str1, str0);
  25.                 console.log(str2); //str2就是{"id":"1","name":"gly","obj":{"id":"32","name":"ffff"}};

  7.                 console.log(str2); //str2就是'true';

  10.                 var str1 = { "id": "1", "name": "gly" };
  11.                 var str0 = { "id": "1", "name": "gly","sex":"男" };

  19.                 console.log(str2); //str2就是'true';
  21.                  //例如4：   
  22.                 var str1 = [{ "id": "1", "name": "gly" }, { "id": "2", "name": "gly" }]; 
  23.                 var str0 = [{ "id": "2", "name": "gly" }, { "id": "1", "name": "gly" }];

  6.                         console.log(str2);
  7.                             //{"name":  "gly" };
  8.                             //值为1的属性全部清理掉
  10.                          //例如2：   
  11.                         var str1 = [{ "id": "1", "name": "gly" }];

  13.                         console.log(str2);
  14.                             //str2就是 [{ }];
  15.                             //所有String类型的全部清理掉
