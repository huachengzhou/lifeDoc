g.RMB Function(Number,String,Number,Number) 
  2.               //*-------------------将数字转化为货币 返回String类型//
  4.                     //例如1：
  5.                     var str1 = g.RMB(11212222.36586, '￥', 2, 3);// 2表示保留的小数，3是千位符
  6.                     console.log(str1);//str1是￥11,212,222.37
  8.                     //例如2：
  9.                     var str1 = g.RMB(11212222.36586);//默认就是上面例子的参数
  10.                     console.log(str1);//str1是￥11,212,222.37
  12.                     //例如3：
  13.                     var str1 = g.RMB(11212222.36586,'$',0);//取整数
  14.                     console.log(str1);//str1是$11,212,222
  17.                     //例如4：
  18.                     var str1 = g.RMB(11212222.36586, '$', 2,0);//不用千位符，保留2为小数
  19.                     console.log(str1);//str1是$11212222.37
g.RMB(Number,String,Number1,Number2)
//String是货币符号￥,Number2表示小数后面保留几位默认为2，Number3表示千分隔默认为3
